## Install

```sh
npm i --save just-test-1
```

## Usage

```
import {Button, ButtonGroup, Icon} from 'just-test-1'
import 'just-test-1/dist/index.css'

export default {
  name: 'app',
  components: {
    'g-button': Button,
    'g-icon': Icon
  }
}
```
